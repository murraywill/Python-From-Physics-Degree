import matplotlib.pyplot as plt
import scipy.optimize as scpo 
import numpy as np

xvalues = np.array([0.000000578, 0.000001156, 0.000001734, 0.000002312, 0.00000289, 0.000003468, 5.46075E-07, 1.09215E-06, 1.63823E-06, 2.1843E-06, 2.73038E-06, 3.27645E-06, 3.82253E-06])
yel_xvalues = np.array([0.000000578, 0.000001156, 0.000001734, 0.000002312, 0.00000289, 0.000003468])
Gre_xvalues = np.array([5.46075E-07, 1.09215E-06, 1.63823E-06, 2.1843E-06, 2.73038E-06, 3.27645E-06, 3.82253E-06])

# y Values for Combined
Set1_yvalues = np.array([0.000328516, 0.000523266, 0.00070225, 0.0009, 0.001105563, 0.001269141, 0.000217563, 0.00034225, 0.000495063, 0.000631266, 0.00081225, 0.000961, 0.001130641])
yel_yvalues = np.array([0.000328516, 0.000523266, 0.00070225, 0.0009, 0.001105563, 0.001269141])

Gre_yvalues = np.array([0.000217563, 0.00034225, 0.000495063, 0.000631266, 0.00081225, 0.000961, 0.001130641])
                       
#Errors in y combined
Set1_yerrors = np.array([2.56326E-05, 3.23501E-05, 3.74767E-05, 4.24264E-05, 4.70226E-05, 5.03814E-05, 2.08597E-05, 2.6163E-05, 3.14663E-05, 3.55321E-05, 4.03051E-05, 4.38406E-05, 4.75529E-05])

#Errors in x combined
Set1_xerrors = np.array([0.000000001, 0.000000002, 0.000000003, 0.000000004, 0.000000005, 0.000000006, 1E-12, 2E-12, 3E-12, 4E-12, 5E-12, 6E-12, 7.00E-12])

                       
                       
                       
#Function for line of best fit combined:
def Line(x, gradient, intercept): 
    return gradient*x + intercept

# Finding the parameters of the line of best fit for Combined:
Set1_actual_fit_parameters, covariance_matrix = scpo.curve_fit(Line, xvalues,Set1_yvalues) 

Set1fit_gradient = Set1_actual_fit_parameters[0] 

Set1fit_intercept = Set1_actual_fit_parameters[1]

# Finding the parameters of the line of best fit for Yellow:                      
                       
Yel_actual_fit_parameters, covariance_matrix = scpo.curve_fit(Line, yel_xvalues,yel_yvalues) 

Yelfit_gradient = Yel_actual_fit_parameters[0] 

Yelfit_intercept = Yel_actual_fit_parameters[1]

# Finding the parameters of the line of best fit for Green:  
                       
Gre_actual_fit_parameters, covariance_matrix = scpo.curve_fit(Line, Gre_xvalues,Gre_yvalues) 

Grefit_gradient = Gre_actual_fit_parameters[0] 

Grefit_intercept = Gre_actual_fit_parameters[1]                       
                                            
# Making the best fit lines:                        
Set1ybestfit = Line(xvalues, Set1fit_gradient, Set1fit_intercept)
Yelybestfit = Line(yel_xvalues, Yelfit_gradient, Yelfit_intercept)
Greybestfit = Line(Gre_xvalues, Grefit_gradient, Grefit_intercept)

# Plotting Residuals 
residuals_YG = np.array([5.32857E-05, 7.85092E-05, 8.7967E-05, 0.00011619, 0.000152226, 0.000146278, -4.83038E-05, -8.37793E-05, -9.11299E-05, -0.00011509, -9.42684E-05, -0.000105681, -9.62038E-05])
residuals_Yel = np.array([1.43304E-06, -3.3808E-06, 7.57098E-06, -2.42857E-07, -1.58692E-05, 1.04888E-05])
residuals_Gre = np.array([-2.19068E-05, 6.76004E-06, 7.3019E-06, 2.44531E-05, -3.1769E-06, 1.42746E-06, -1.48588E-05])


plt.figure(figsize=(8, 5))
plt.scatter(xvalues, residuals_YG, color='blue', marker='o' , label="Residuals Combined")
plt.scatter(yel_xvalues, residuals_Yel, color='orange', marker='s' , label="Residuals Yellow")
plt.scatter(Gre_xvalues, residuals_Gre, color='green', marker='x' , label="Residuals Green")



plt.figure(1)

# Y errors Yellow
plt.errorbar(xvalues, Set1_yvalues, xerr = Set1_xerrors , yerr = Set1_yerrors , fmt = 'o')


# This line adds the x axis label "x values":
plt.xlabel("Fringe Number times Wavelength")

# This line adds the y axis label "y values":
plt.ylabel("Radius of Fringe squared")
plt.plot(xvalues, Set1ybestfit)
plt.plot(yel_xvalues, Yelybestfit)
plt.plot(Gre_xvalues, Greybestfit)

plt.show()
